Supabase Integration

Connect your Feed Mill report to a Supabase database to persist silo metrics in real-time.

Database Setup Instructions

Execute this SQL in your Supabase SQL Editor to create the tables:

-- Create materials table
create table if not exists public.materials (
  name text primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create silos table
create table if not exists public.silos (
  id integer primary key,
  name text not null,
  status text not null,
  run_time numeric not null default 0,
  current_moisture numeric not null default 0,
  purchase_moisture numeric not null default 0,
  fan_status text not null,
  material_type text references public.materials(name) on update cascade on delete set null,
  capacity numeric not null,
  fill_level numeric not null,
  current_fill_tons numeric not null,
  filling_start text not null,
  filling_end text not null,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Insert default materials
insert into public.materials (name) 
values ('Maize'), ('Rice')
on conflict (name) do nothing;
