Supabase Integration

Connect your Feed Mill report to a Supabase database to persist silo metrics in real-time.

Database Setup Instructions

Execute this SQL in your Supabase SQL Editor to create the tables:

-- Create materials table
create table if not exists public.materials (
  name text primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create silos table
create table if not exists public.silos (
  id integer primary key,
  name text not null,
  status text not null,
  run_time numeric not null default 0,
  current_moisture numeric not null default 0,
  purchase_moisture numeric not null default 0,
  fan_status text not null,
  material_type text references public.materials(name) on update cascade on delete set null,
  capacity numeric not null,
  fill_level numeric not null,
  current_fill_tons numeric not null,
  filling_start text not null,
  filling_end text not null,
  fan_on_time text not null default '-',
  fan_off_time text not null default '-',
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Insert default materials
insert into public.materials (name) 
values ('Maize'), ('Rice')
on conflict (name) do nothing;

-- Create daily_maize_logs table
create table if not exists public.daily_maize_logs (
  id bigint primary key,
  date text not null,
  silo_number text not null,
  purchase_moisture numeric not null default 0,
  formula_moisture numeric not null default 0,
  c_room_un_grind numeric[],
  lab_wet_un_grind numeric[],
  lab_wet_grind numeric[],
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create daily_less_excess_logs table
create table if not exists public.daily_less_excess_logs (
  id bigint primary key,
  report_date text not null,
  shift text,
  officer_name text,
  feed_name text,
  batches numeric not null default 0,
  production_bags numeric not null default 0,
  water_addition numeric not null default 0,
  remarks text,
  is_locked boolean not null default false,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create five_s_logs table
create table if not exists public.five_s_logs (
  id bigint primary key,
  audit_date text not null,
  auditor_name text not null,
  area_name text not null,
  sort_score numeric not null default 5,
  set_order_score numeric not null default 5,
  shine_score numeric not null default 5,
  standardize_score numeric not null default 5,
  sustain_score numeric not null default 5,
  remarks text not null default '',
  is_locked boolean not null default false,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- Create daily_checklists table
create table if not exists public.daily_checklists (
  id bigint primary key,
  checklist_date text not null,
  department_name text not null,
  filled_by text not null,
  checked_items boolean[] not null,
  remarks text not null default '',
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null,
  constraint uniq_date_dept unique(checklist_date, department_name)
);

-- Create shift_reports table
create table if not exists public.shift_reports (
  id bigint primary key,
  report_date text not null,

  feed_produced text not null default '',

  is_submitted boolean not null default false,

-- Create daily less/excess report view
create or replace view public.v_daily_less_excess_report as
with calculations as (
  select
    id,
    report_date,
    shift,
    officer_name,
    feed_name,
    batches,
    production_bags,
    water_addition,
    remarks,
    is_locked,
    updated_at,
    (batches * 100) as expected_bags,
    (production_bags - (batches * 100)) as diff_bags
  from
    public.daily_less_excess_logs
)
select
  id,
  report_date,
  shift,
  officer_name,
  feed_name,
  batches,
  production_bags,
  expected_bags,
  diff_bags,
  case 
    when expected_bags > 0 then round((diff_bags::numeric / expected_bags::numeric) * 100, 2)
    else 0.00
  end as diff_pct,
  water_addition,
  remarks,
  is_locked,
  updated_at,
  sum(expected_bags) over (partition by feed_name order by id asc) as cumulative_expected_bags,
  sum(diff_bags) over (partition by feed_name order by id asc) as cumulative_diff_bags,
  case
    when sum(expected_bags) over (partition by feed_name order by id asc) > 0 
      then round(
        (sum(diff_bags) over (partition by feed_name order by id asc)::numeric / 
         sum(expected_bags) over (partition by feed_name order by id asc)::numeric) * 100, 
        2
      )
    else 0.00
  end as overall_avg_less_excess_pct
from
  calculations
order by
  id desc;

-- Create daily maize moisture report view
create or replace view public.v_daily_maize_moisture_report as
with averages as (
  select
    id,
    date,
    silo_number,
    purchase_moisture,
    formula_moisture,
    c_room_un_grind,
    lab_wet_un_grind,
    lab_wet_grind,
    updated_at,
    -- Average of control room unground moisture readings (ignoring nulls)
    (select round(avg(val)::numeric, 1) from unnest(c_room_un_grind) as val) as c_room_un_grind_avg,
    -- Average of lab wet unground moisture readings (ignoring nulls)
    (select round(avg(val)::numeric, 1) from unnest(lab_wet_un_grind) as val) as lab_wet_un_grind_avg,
    -- Average of lab wet ground moisture readings (ignoring nulls)
    (select round(avg(val)::numeric, 1) from unnest(lab_wet_grind) as val) as lab_wet_grind_avg
  from
    public.daily_maize_logs
)
select
  id,
  date,
  silo_number,
  purchase_moisture,
  formula_moisture,
  c_room_un_grind,
  c_room_un_grind_avg,
  lab_wet_un_grind,
  lab_wet_un_grind_avg,
  lab_wet_grind,
  lab_wet_grind_avg,
  -- Average Consumed Moisture is equivalent to control room average
  c_room_un_grind_avg as avg_consumed_moisture,
  -- Difference = Running Average Consumed Moisture - Formula Moisture
  case 
    when c_room_un_grind_avg is not null and formula_moisture is not null 
      then round((c_room_un_grind_avg - formula_moisture)::numeric, 1)
    else null
  end as moisture_difference,
  updated_at
from
  averages
order by
  id desc;

-- Create five_s_report view
create or replace view public.v_five_s_report as
select
  id,
  audit_date,
  auditor_name,
  area_name,
  sort_score,
  set_order_score,
  shine_score,
  standardize_score,
  sustain_score,
  (sort_score + set_order_score + shine_score + standardize_score + sustain_score) as total_score,
  round((sort_score + set_order_score + shine_score + standardize_score + sustain_score)::numeric / 5.0, 2) as average_score,
  remarks,
  is_locked,
  updated_at
from
  public.five_s_logs
order by
  id desc;
